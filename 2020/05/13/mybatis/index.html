mybatis使用手册

MyBatis 是一款优秀的持久层框架，它支持自定义 SQL、存储过程以及高级映射。MyBatis 免除了几乎所有的 JDBC 代码以及设置参数和获取结果集的工作。MyBatis 可以通过简单的 XML 或注解来配置和映射原始类型、接口和 Java POJO（Plain Old Java Objects，普通老式 Java 对象）为数据库中的记录。

        select  * from tb_user where id= #{id}

</mapper>

  • 新建一个映射的配置文件UserDaoMapper.xml
    <?xml version="1.0" encoding="UTF-8" ?>
<!DOCTYPE mapper
        PUBLIC "-//mybatis.org//DTD Mapper 3.0//EN"
        "http://mybatis.org/dtd/mybatis-3-mapper.dtd">
<mapper namespace="userdao">

<!--    查询因为有结果集，要制定返回类型-->
<!--    #{id}里面的名字任意，只要保证唯一-->
    <select id="queryUserById" resultType="com.kaidow.mybatis.User">
        select * from tb_user where id= #{id}
    </select>

    <select id="queryAllUsers" resultType="com.kaidow.mybatis.User">
        select * from tb_user
    </select>

    <insert id="insertUser" >
        insert into tb_user(
        id,
        username,
        password,
        age,
        sex,
        birthday,
        created,
        updated
        )
        values (
        #{id},
        #{username},
        #{password},
        #{age},
        #{sex},
        #{birthday},
        now(),
        now()
        );
    </insert>

<!--    suffixoverride去除多余的后缀，也就是最后一个“，”-->
    <update id="updateUser" >
        update tb_user
        <!-- 这里也可以用set标签 --!>
        <trim prefix="set" suffixOverrides=",">

            <if test="username!=null">username=#{username},</if>
            <if test="password!=null">password=#{password},</if>
            <if test="age!=null">age=#{age},</if>
            <if test="sex!=null">sex=#{sex},</if>
            <if test="birthday!=null">birthday=#{birthday},</if>
            created=now(),
            updated=now()
        </trim>
        <where>
            id=#{id}
        </where>
    </update>

    <delete id="deleteUser" >
        delete from tb_user where id=#{id};
    </delete>

</mapper>
  • 创建一个实现类UserDaoImpl   这里如果报错Mapped Statement collection does not contain value for mapper很可能是参数列表里的命名空间.id与mapperxml文件里的不一致
    public class UserDaoImpl implements UserDao{
    private SqlSession sqlSession;

    public UserDaoImpl(SqlSession sqlSession) {
        this.sqlSession = sqlSession;
    }

    @Override
    public User queryUserById(String id) {
        return this.sqlSession.selectOne("userdao.queryUserById",id);
    }

    @Override
    public List<User> queryAllUsers() {
        return this.sqlSession.selectList("userdao.queryAllUsers");
    }

    @Override
    public void insertUser(User u) {
        this.sqlSession.insert("userdao.insertUser",u);
    }

    @Override
    public void deleteUser(String id) {
        sqlSession.delete("userdao.deleteUser",id);
    }

    @Override
    public void updateUser(User u) {
        sqlSession.update("userdao.updateUser",u);
    }
}

    3.模糊查询与多条件查询

    <!--%表示任意字符，#{0}里面的id任意取-->
    <select id="matchSelect" parameterType="string" resultType="com.kaidow.mybatis.User">
        select * from tb_user where username like concat('%',#{0},'%')
    </select>
<!--这里参数是两个不同类型的，用map-->
    <select id="multiConditionSelect" parameterType="map" resultType="com.kaidow.mybatis.User">
        select * from tb_user where username like concat('%',#{name},'%') and age>#{age}
    </select>

  • 新建映射文件category.xml
    <mapper namespace="category">
    <!-- id唯一标识，type为映射的pojo对象-->
    <resultMap id="CategoryBean" type="com.kaidow.mybatis.pojo.Category">
<!--        id column为表中主键字段，property为pojo中对应的属性名-->
        <id column="cid" property="id"/>
        <result column="cname" property="name"/>

        <collection property="products" ofType="com.kaidow.mybatis.pojo.Product">
            <id column="pid" property="id"/>
            <result column="pname" property="name"/>
            <result column="price" property="price"/>
        </collection>
    </resultMap>
<!--    表的列名要重新命名与上面的column一致，从而与pojo中的属性对应-->
    <select id="listCategory" resultMap="CategoryBean">
        select c.*,p.* ,c.id cid,c.name cname,p.id pid,p.name pname
        from ssmdemo.category c left join product p on c.id = p.cid
    </select>
</mapper>

5.多对一

多个product可以对应一个category，那么product类中就要有一个category属性，从而可以根据每个product得到对应的category

  • product类中添加category属性
    private int id;
    private String name;
    private float price;
    private Category category;//添加的category
  • 新建product.xml,通过product2category的select标签配置sql语句，通过resultmap返回查询的结果集，进行字段与属性的对应
    <resultMap id="productBean" type="com.kaidow.mybatis.pojo.Product">
        <id column="pid" property="id"/>
        <result column="pname" property="name"/>
        <result column="price" property="price"/>
        <association property="category" javaType="com.kaidow.mybatis.pojo.Category">
            <id column="cid" property="id"/>
            <result column="cname" property="name"/>
        </association>
    </resultMap>

    <select id="product2category" resultMap="productBean">
        select p.*,c.*,p.id pid,p.name pname,c.id cid,c.name cname
        from product p left join category c on p.cid=c.id
    </select>

    6.多对多

    为了测试多对多，需要换一种情境。订单Order类，一个订单有多个订单项OrderItem，每一个订单项对应一种订单产品Product。因而，一个订单可以对应多个产品，一个产品也可以对应多个订单，也就是一个产品可以在多个订单中出现。
